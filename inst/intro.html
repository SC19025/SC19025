rw.Metropolis <- function( sigma, x0, N) {
x <- numeric(N)
x[1] <- x0
u <- runif(N)
k <- 0
for (i in 2:N) {
y <- rnorm(1, x[i-1], sigma)
if (u[i] <= exp(-abs(y)) / exp(-abs(x[i-1]))  ){
x[i] <- y
}
else {
x[i] <- x[i-1]
k <- k + 1
}
}
return(list(x=x, k=k))
}
